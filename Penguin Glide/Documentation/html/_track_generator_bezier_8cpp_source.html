119 void ATrackGeneratorBezier::CreateVertices(TArray<FVector>& points, TArray<FVector>& vertices, TArray<FVector>& normals) {
120  std::vector<FVector> verts(arraySize * 2);
121  FVector firstControlPoint;
122  FVector secondControlPoint;
123  FVector pivot;
124 
125  int vertAmount = (triangleAmount / 2) + 1;
126  int vertIndex = 0;
127 
128  for (int i = 0; i < arraySize; i++)
129  {
130  FVector forwardDirection = FVector::ZeroVector;
131  if (i < arraySize - 1)
132  {
133  forwardDirection += points[i + 1] - points[i];
134  }
135 
136  if (i > 0)
137  {
138  forwardDirection += points[i] - points[i - 1];
139  }
140  forwardDirection.Normalize();
141 
142  FVector left = FVector(-forwardDirection.Y, forwardDirection.X, 0);
143 
144  float trackWidth = 100.0f;
145 
146  verts[vertIndex] = points[i] + left * trackWidth;
147  verts[vertIndex].Z += zPosition;
148  verts[vertIndex + 1] = points[i] - left * trackWidth;
149  verts[vertIndex + 1].Z += zPosition;
150  zPosition -= steepness;
151 
152  int trackDepth = 80;
153  firstControlPoint = verts[vertIndex] - FVector(0, 0, trackDepth);
154  secondControlPoint = verts[vertIndex + 1] - FVector(0, 0, trackDepth);
155 
156  float t = 0.0f;
157 
158  pivot = (verts[vertIndex] + (verts[vertIndex + 1] - verts[vertIndex]) / 2);
159 
160  tempPlayer->AddTrackPivots(pivot * trackScale);
161 
162  for (int x = 0; x < vertAmount; x++)
163  {
164  //use cubic B�zier curve algorithm to calculate points on a curve in between two points
165  FVector vertexPosition = pow((1 - t), 3) * verts[vertIndex] + (3 * t) * pow((1 - t), 2) * firstControlPoint
166  + (3 * pow(t, 2)) * (1 - t) * secondControlPoint + pow(t, 3) * verts[vertIndex + 1];
167 
168  if (i == 0 && endingVertsInitialized)
169  {
170  vertices.Add(endingVerts[x]); //set the first row of vertices equal to the last row of the previous mesh section
171  if (x == vertAmount - 1)
172  {
173  endingVerts.Empty(); //empty endingVerts, to make room of the endingVerts of current mesh section
174  }
175  }
176  else
177  {
178  vertices.Add(vertexPosition);
179  }
180  if (i == arraySize - 1)
181  {
182  endingVerts.Add(vertexPosition); //last row of mesh section, add vertices to endingVerts array
183  }
184 
185  t += 1 / ((float)vertAmount - 1); //increase t to shift to the next point on the curve
186  FVector normal = pivot - vertices[x + vertAmount * i];
187  float normalLength;
188  normalLength = sqrtf(powf(normal.X, 2) + powf(normal.Y, 2) + powf(normal.Z, 2));
189  normal /= normalLength;
190  normals.Add(normal);
191  }
192  vertIndex += 2;
193  }
194 
195  int sizeVertices = vertices.Num();
196  for (int i = 0; i < sizeVertices - 1; i++)
197  {
198  vertices[i].Z -= yPerSectBase * yPerSection;
199  }
200 }
